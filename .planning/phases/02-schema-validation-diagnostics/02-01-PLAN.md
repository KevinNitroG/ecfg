```go
type ValidationError struct {
    Property string        // Property name (e.g., "indent_style")
    Value    string        // Invalid value
    Reason   string        // Human-readable reason
    Range    parser.Range  // Position from AST for diagnostic
}

func Validate(doc *parser.Document) []ValidationError {
    var errors []ValidationError
    
    // Validate preamble
    if doc.Preamble != nil {
        for _, kv := range doc.Preamble.KeyValues {
            if err := validateProperty(kv, true); err != nil {
                errors = append(errors, *err)
            }
        }
    }
    
    // Validate sections
    for _, section := range doc.Sections {
        for _, kv := range section.KeyValues {
            if err := validateProperty(kv, false); err != nil {
                errors = append(errors, *err)
            }
        }
    }
    
    return errors
}

func validateProperty(kv *parser.KeyValue, inPreamble bool) *ValidationError {
    // 1. Check if property exists in Schema
    // 2. Check PreambleOnly constraint (SCHEMA-02)
    // 3. Validate value based on PropertyType:
    //    - Enum: value in ValidValues
    //    - Integer: parse and check Min/Max bounds
    //    - Boolean: value is "true" or "false"
    //    - String: accept any value
    // 4. Check SpecialValues for properties like indent_size="tab"
    // 5. Return ValidationError with kv.ValueRange if validation fails
}
```

**Implementation Notes:**
- Use `strconv.Atoi()` for integer parsing
- Use `strings.ToLower()` for case-insensitive property names
- ValidationError.Range should use KeyValue.ValueRange (not full Range) for precise underline
- For PreambleOnly errors, use KeyValue.KeyRange to underline the property name

Run tests until all pass.
