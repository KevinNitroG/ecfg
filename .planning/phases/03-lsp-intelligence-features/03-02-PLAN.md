**Implement completion context detection:**

The challenge: Determine if user is typing a property key (before `=`) or a property value (after `=`).

**Strategy:** 
1. Try to find existing KeyValue node at position using FindNodeAtPosition
2. If found:
   - PartKey → completing key (suggest property names)
   - PartValue → completing value (suggest valid values for that property)
3. If not found (position not in existing KeyValue):
   - Assume completing a new property key

```go
type CompletionContext struct {
    CompletingKey   bool     // Typing property key
    CompletingValue bool     // Typing property value
    PropertyKey     string   // For value completion, which property?
    InPreamble      bool
    InSection       bool
}

func detectCompletionContext(doc *parser.Document, pos protocol.Position) CompletionContext {
    ctx := CompletionContext{}
    
    // Try to resolve to existing node
    node := FindNodeAtPosition(doc, pos)
    
    if node == nil {
        // Not on existing node - assume typing new property key
        ctx.CompletingKey = true
        
        // Determine if in preamble or section by checking position against ranges
        if doc.Preamble != nil && containsPosition(doc.Preamble.Range, lspPositionToParser(pos)) {
            ctx.InPreamble = true
        } else {
            // Check which section contains this position
            for _, section := range doc.Sections {
                if containsPosition(section.Range, lspPositionToParser(pos)) {
                    ctx.InSection = true
                    break
                }
            }
        }
        
        return ctx
    }
    
    // On existing KeyValue node
    if node.Part == PartKey {
        ctx.CompletingKey = true
        ctx.PropertyKey = node.KeyValue.Key  // User might be editing existing key
    } else if node.Part == PartValue {
        ctx.CompletingValue = true
        ctx.PropertyKey = node.KeyValue.Key  // Complete values for this property
    }
    
    ctx.InPreamble = node.InPreamble
    ctx.InSection = node.InSection
    
    return ctx
}
```

**Implement completePropertyKeys function:**

```go
func completePropertyKeys(inPreamble bool) []protocol.CompletionItem {
    items := []protocol.CompletionItem{}
    
    for name, schema := range validator.Schema {
        // Filter out root if not in preamble
        if schema.PreambleOnly && !inPreamble {
            continue
        }
        
        detail := schema.Type.String()
        
        // Add insert text with " = " suffix for convenience
        insertText := name + " = "
        
        item := protocol.CompletionItem{
            Label:            name,
            Kind:             protocol.CompletionItemKindProperty,
            Detail:           &detail,
            Documentation: &protocol.MarkupContent{
                Kind:  protocol.Markdown,
                Value: schema.Description,
            },
            InsertText:       &insertText,
            InsertTextFormat: ptrInsertTextFormat(protocol.InsertTextFormatPlainText),
        }
        
        items = append(items, item)
    }
    
    // Sort alphabetically for consistent ordering
    sort.Slice(items, func(i, j int) bool {
        return items[i].Label < items[j].Label
    })
    
    return items
}

func ptrInsertTextFormat(f protocol.InsertTextFormat) *protocol.InsertTextFormat {
    return &f
}
```

```go
func completePropertyValues(propertyKey string) []protocol.CompletionItem {
    // Case-insensitive lookup
    propertyKey = strings.ToLower(strings.TrimSpace(propertyKey))
    
    schema, exists := validator.Schema[propertyKey]
    if !exists {
        return []protocol.CompletionItem{}  // Unknown property
    }
    
    items := []protocol.CompletionItem{}
    
    // For integer properties, don't suggest completions (user types number)
    if schema.Type == validator.PropertyTypeInteger {
        // But do suggest special values if any exist
        if len(schema.SpecialValues) == 0 {
            return []protocol.CompletionItem{}
        }
    }
    
    // Add enum values
    for _, value := range schema.ValidValues {
        detail := fmt.Sprintf("Valid value for %s", propertyKey)
        item := protocol.CompletionItem{
            Label:      value,
            Kind:       protocol.CompletionItemKindValue,
            Detail:     &detail,
            InsertText: &value,
        }
        items = append(items, item)
    }
    
    // Add special values (e.g., "tab" for indent_size, "off" for max_line_length)
    for _, value := range schema.SpecialValues {
        detail := fmt.Sprintf("Special value for %s", propertyKey)
        item := protocol.CompletionItem{
            Label:      value,
            Kind:       protocol.CompletionItemKindValue,
            Detail:     &detail,
            InsertText: &value,
        }
        items = append(items, item)
    }
    
    return items
}
```

**Implement main ComputeCompletion function:**

```go
func ComputeCompletion(doc *parser.Document, pos protocol.Position) *protocol.CompletionList {
    ctx := detectCompletionContext(doc, pos)
    
    var items []protocol.CompletionItem
    
    if ctx.CompletingKey {
        items = completePropertyKeys(ctx.InPreamble)
    } else if ctx.CompletingValue {
        items = completePropertyValues(ctx.PropertyKey)
    } else {
        // Unknown context, return empty
        items = []protocol.CompletionItem{}
    }
    
    return &protocol.CompletionList{
        IsIncomplete: false,  // EditorConfig is small, always return all
        Items:        items,
    }
}
```

```go
func TestCompleteAllProperties(t *testing.T) {
    tests := []struct {
        property      string
        inPreamble    bool
        shouldInclude bool
    }{
        {"root", true, true},
        {"root", false, false},  // root excluded from sections
        {"indent_style", true, true},
        {"indent_style", false, true},
        {"indent_size", true, true},
        {"tab_width", true, true},
        {"end_of_line", true, true},
        {"charset", true, true},
        {"trim_trailing_whitespace", true, true},
        {"insert_final_newline", true, true},
        {"max_line_length", true, true},
    }
    
    for _, tt := range tests {
        t.Run(fmt.Sprintf("%s_preamble=%v", tt.property, tt.inPreamble), func(t *testing.T) {
            var source string
            var position protocol.Position
            
            if tt.inPreamble {
                source = "x"
                position = protocol.Position{Line: 0, Character: 1}
            } else {
                source = "[*.go]\nx"
                position = protocol.Position{Line: 1, Character: 1}
            }
            
            doc := parser.Parse(source)
            completion := ComputeCompletion(doc, position)
            
            found := false
            for _, item := range completion.Items {
                if item.Label == tt.property {
                    found = true
                    break
                }
            }
            
            if tt.shouldInclude {
                assert.True(t, found, "%s should be in completion list", tt.property)
            } else {
                assert.False(t, found, "%s should NOT be in completion list", tt.property)
            }
        })
    }
}
```
